\begin{tikzpicture}[scale=1.33,every node/.style={ellipse, draw}]
    % Hyperparameters
    \node (ay) at (0, 5) {$a_Y$};
    \node (by) at (1, 5) {$b_Y$};
    \node (sy) at (2, 5) {$\sigma_Y$};
    \node (ma) at (3, 5) {$\mu_\alpha$};
    \node (sa) at (4, 5) {$\sigma_\alpha$};

    % Input parameters
    \node (mi) at (0, 4) {$M_i$};
    \node (yi) at (1, 4) {$Y_i$} edge [<-] (ay) edge [<-] (by) edge [<-] (sy);
    \node (fehi) at (2, 4) {$\mathrm{[Fe/H]}_i$} edge [->] (yi);
    \node (ai) at (3, 4) {$\alpha_i$} edge [<-] (ma) edge [<-] (sa);
    \node (eepi) at (4, 4) {$\mathrm{EEP}_i$};

    % Output parameters
    \node (age) at (0, 3) {$t_i$};
    \node [fill=black!33] (teff) at (1, 3) {$T_{\mathrm{eff}, i}$};
    \node [fill=black!33] (lum) at (2, 3) {$L_i$} edge [<-] (teff);
    \node (rad) at (3, 3) {$R_i$} edge [->] (lum);
    \node (dnu) at (4, 3) {$\Delta\nu_i$};

    % Connect neural network inputs and outputs
    \foreach \x in {mi, yi, fehi, ai, eepi}
    \foreach \y in {age, teff, rad, dnu}
    \draw[->] (\x) -- (\y);
\end{tikzpicture}
